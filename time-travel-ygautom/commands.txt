# ------------------------------
# Section 1 – Git Commands
# ------------------------------

# Repository initialization and setup
git init
git checkout -b main
git add .
git commit -m "chore: initial repo setup"

# Remote linking and pushing
git remote add origin https://github.com/Ygautom/time-travel-ygautom.git
git push -u origin main

# Branching
git branch author/idea-sprint
git branch editor/review
git checkout author/idea-sprint
git checkout editor/review

# Creating feature branches
git branch feat/layout
git branch feat/content
git branch feat/polish

# Adding and committing files
git add README.md
git add LICENSE CODE_OF_CONDUCT.md
git add RELEASE_NOTES.md GIT_JOURNAL.md
git commit -m "docs: add project documentation files"

# Merging and resolving conflicts
git merge author/idea-sprint
git merge editor/review
# (Resolved merge conflict manually in zine/intro.md)

# Tagging releases
git tag v0.1
git tag -a v1.0 -m "Final release"

# Pushing branches and tags
git push -u origin author/idea-sprint
git push -u origin editor/review
git push origin --tags

git status
git log --oneline --graph

git diff

# ------------------------------
# Section 2 – Maven Commands
# ------------------------------

# Verify Maven installation
mvn -v

# Create new project
mvn archetype:generate -DgroupId=edu.asu.ygautom \
-DartifactId=ygautom-Maven-Practice \
-DarchetypeArtifactId=maven-archetype-quickstart \
-DarchetypeVersion=1.4 -DinteractiveMode=false

# Build lifecycle
mvn compile
mvn test
mvn clean
mvn package

# Verify output
ls target/

# Section 3 – Graphviz Commands

dot -V
